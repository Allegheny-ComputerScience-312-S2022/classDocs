/* inner (left) join */
SELECT tableA.num
FROM tableB
LEFT JOIN tableA
ON tableA.num == tableB.num;


/* inner (left) join */
SELECT distinct(tableA.num)
FROM tableB
LEFT JOIN tableA
ON tableA.num == tableB.num;

/* inner (right) join */
SELECT tableB.num
FROM tableA
LEFT JOIN tableB
ON tableA.num == tableB.num;
