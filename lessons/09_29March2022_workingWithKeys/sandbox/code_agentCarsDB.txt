/* Enable foreign keys: automatically check any foreign key constraints */
PRAGMA foreign_keys = ON;

DROP TABLE Car;
CREATE TABLE Car (
name VARCHAR PRIMARY KEY,
registration VARCHAR,
capacity INT,
topSpeed INT
);

DROP TABLE Agent;
CREATE TABLE Agent (
id INT PRIMARY KEY,
lastName VARCHAR,
registration VARCHAR,
worksFor VARCHAR,
FOREIGN KEY(registration) REFERENCES car(name));
